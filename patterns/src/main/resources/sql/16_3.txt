    USERS_ID_ASSIGNEDTO BIGINT UNSIGNED NOT NULL,
    FOREIGN KEY (ISSUESLISTS_ID) REFERENCES ISSUESLISTS(ID),
    FOREIGN KEY (USERS_ID_ASSIGNEDTO) REFERENCES USERS(ID)
);
