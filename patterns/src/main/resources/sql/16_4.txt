SELECT I.ID, I.SUMMARY, IL.NAME
FROM ISSUESS I JOIN ISSUESLISTS IL ON I.ISSUESLISTS_ID = IL.ID;

SELECT I.ID, I.SUMMARY, U.ID
FROM ISSUESS I JOIN USERS U ON I.USERS_ID_ASSIGNEDTO = U.ID;

SELECT U.FIRSTNAME, U.LASTNAME, COUNT(*) AS ISSUES_NUMBER
FROM USERS U JOIN ISSUESS I ON U.ID = I.USERS_ID_ASSIGNEDTO
GROUP BY I.USERS_ID_ASSIGNEDTO
HAVING COUNT(*) > 0;
